Add an admin on the smart wallet account.

```jsx
const Component = () => {
  const { mutate: addAdmin, isLoading, error } = useAddAdmin();

  if (error) {
    console.error("failed to add admin", error);
  }

  return (
    <button disabled={isLoading} onClick={() => addAdmin("0x...")}>
      Add admin
    </button>
  );
};
```

## Usage

Provide the smart wallet account contract address as the argument.

```jsx
import { useAccountAdminsAndSigners } from "@thirdweb-dev/react";

// Your ERC20 token smart contract address
const adminAddress = "{{admin_address}}";

const Component = () => {
  const { mutate: addAdmin, isLoading, error } = useAddAdmin();

  if (error) {
    console.error("failed to add admin", error);
  }

  return (
    <button disabled={isLoading} onClick={() => addAdmin(adminAddress)}>
      Add admin
    </button>
  );
};
```
